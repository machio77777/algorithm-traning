Node * find(Node *u, int k) {
    while (u != NIL && k != u->key) {
        // уЈЙтюеУф┐сЂ╣сЂдсЂёсѓІу»ђуѓ╣uсЂ«сѓГсЃ╝сѓѕсѓіСИјсЂѕсѓЅсѓїсЂЪсѓГсЃ╝kсЂїт░ЈсЂЋсЂёта┤тљѕсЂ»тидУдЂу┤асЂИуД╗тІЋ
        if (k < u->key) {
            u = u->left;

        } else {
            u = u->right;
        }
    }
    return u;
}
