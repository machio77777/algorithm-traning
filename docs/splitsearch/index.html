int S[] = {1, 2, 3, 5, 6, 7, 8, 11, 19, 20, 44, 90};

int binarySearch(int key) {
    
    int left = 0;
    int right = n;
    int mid;
    
    while (left < right) {
        mid = (left + right) / 2;
        
        // СИГжќЊУдЂу┤асЂеkeyсЂїтљїСИђсЂ«та┤тљѕ
        if (key == S[mid]) {
            return 1;
        // тЙїтЇіжЃетѕєсѓњТцюу┤бсЂЋсЂЏсѓІсѓѕсЂєтѕХтЙА
        } else if (key > S[mid]) {
            left = mid + 1;
        // тЅЇтЇіжЃетѕєсѓњТцюу┤бсЂЋсЂЏсѓІсѓѕсЂєтѕХтЙА
        } else {
            right = mid;
        }
    }
    return 0;
}
